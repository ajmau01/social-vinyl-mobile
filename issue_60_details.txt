## Problem
Currently, there are no automated end-to-end (E2E) tests to verify critical user journeys work correctly on real devices. While unit and component tests exist, they cannot catch:
- Navigation flow issues
- Integration problems between screens
- Native module failures
- Real WebSocket connection issues
- Actual sync/database operations
- Cross-component state management bugs

**Missing Coverage:**
- ❌ User login flow
- ❌ Collection sync process
- ❌ Search and filtering
- ❌ Release detail viewing
- ❌ View mode switching (artist/genre/decade)
- ❌ Listening bin management
- ❌ Session joining/leaving

## Implementation Approach

### 1. Install Detox
```bash
npm install --save-dev detox detox-cli
npm install --save-dev jest-circus
```

#### **e2e/login.e2e.ts**
```typescript
import { by, device, element, expect as detoxExpect } from 'detox';

describe('Login Flow', () => {
  beforeEach(async () => {
    await device.reloadReactNative();
  });

  it('should display login screen on first launch', async () => {
    await detoxExpect(element(by.id('login-username-input'))).toBeVisible();
    await detoxExpect(element(by.id('login-password-input'))).toBeVisible();
    await detoxExpect(element(by.id('login-submit-button'))).toBeVisible();
  });

  it('should show error for invalid credentials', async () => {
    await element(by.id('login-username-input')).typeText('wronguser');
    await element(by.id('login-password-input')).typeText('wrongpass');
    await element(by.id('login-submit-button')).tap();
    
    await detoxExpect(element(by.text('Invalid credentials'))).toBeVisible();
  });

  it('should successfully login with valid credentials', async () => {
    await element(by.id('login-username-input')).typeText('testuser');
    await element(by.id('login-password-input')).typeText('testpass');
    await element(by.id('login-submit-button')).tap();
    
    // Should navigate to collection screen
    await detoxExpect(element(by.id('collection-screen'))).toBeVisible();
  });

  it('should persist login across app restarts', async () => {
    // First login
    await element(by.id('login-username-input')).typeText('testuser');
    await element(by.id('login-password-input')).typeText('testpass');
    await element(by.id('login-submit-button')).tap();
    
    await detoxExpect(element(by.id('collection-screen'))).toBeVisible();
    
    // Restart app
    await device.reloadReactNative();
    
    // Should still be on collection screen (not login)
    await detoxExpect(element(by.id('collection-screen'))).toBeVisible();
  });
});
```

#### **e2e/collection-sync.e2e.ts**

```typescript
import { by, device, element, expect as detoxExpect, waitFor } from 'detox';

describe('Collection Sync', () => {
  beforeEach(async () => {
    await device.reloadReactNative();
    // Assume user is logged in (use beforeAll login helper)
  });

  it('should sync collection when sync button pressed', async () => {
    await element(by.id('collection-sync-button')).tap();
    
    // Should show syncing status

      .toBeVisible()
      .withTimeout(2000);

      .toBeVisible()
      .withTimeout(30000);

    await detoxExpect(element(by.id('collection-list'))).toBeVisible();

    await element(by.id('collection-sync-button')).tap();
    
    await waitFor(element(by.id('collection-header-progress-bar')))

      .withTimeout(2000);

  it('should disable sync button while syncing', async () => {
    await element(by.id('collection-sync-button')).tap();
    
    // Button should be disabled
    await detoxExpect(element(by.id('collection-sync-button'))).toHaveToggleValue(false);
```

#### **e2e/collection-navigation.e2e.ts**
```typescript
import { by, element, expect as detoxExpect } from 'detox';

describe('Collection Navigation', () => {
  beforeEach(async () => {
    // Ensure collection is synced
    await element(by.id('collection-sync-button')).tap();
    await waitFor(element(by.text('Synced'))).toBeVisible().withTimeout(30000);
  });

  it('should switch between view modes', async () => {
    // Default is artist mode
    await detoxExpect(element(by.id('view-mode-artist'))).toBeVisible();
    
    // Switch to genre mode
    await element(by.id('view-mode-genre')).tap();
    await detoxExpect(element(by.text(/genre/i))).toBeVisible();
    
    // Switch to decade mode
    await element(by.id('view-mode-decade')).tap();
    await detoxExpect(element(by.text(/1960s|1970s|1980s/i))).toBeVisible();
  });

  it('should open release details when release tapped', async () => {
    // Tap first release card
    await element(by.id('collection-list')).atIndex(0).tap();
    
    // Modal should open
    await detoxExpect(element(by.id('release-details-modal'))).toBeVisible();
    await detoxExpect(element(by.id('modal-close-button'))).toBeVisible();
  });

  it('should close release modal when close button tapped', async () => {
    await element(by.id('collection-list')).atIndex(0).tap();
    await detoxExpect(element(by.id('release-details-modal'))).toBeVisible();
    
    await element(by.id('modal-close-button')).tap();
    await detoxExpect(element(by.id('release-details-modal'))).not.toBeVisible();
```

### 6. Setup Mock Backend for Tests

**e2e/helpers/mockServer.ts**
```typescript
import { Server } from 'mock-socket';

export class MockBackendServer {
  private server: Server | null = null;

  start(port: number = 8080) {
    this.server = new Server(\`http://localhost:\${port}\`);
    
    this.server.on('connection', (socket) => {
      console.log('[MockServer] Client connected');
      
      socket.on('message', (data) => {
        const message = JSON.parse(data as string);
        
        // Mock login response
        if (message.type === 'LOGIN') {
          socket.send(JSON.stringify({
            type: 'login-result',
            success: message.username === 'testuser',
            token: 'mock-token-12345'
          }));
        }
        
        // Mock collection data
        if (message.type === 'SYNC') {
          socket.send(JSON.stringify({
            type: 'collection-data',
            albums: { /* mock albums */ }
          }));
        }
      });
    });
  }

  stop() {
    this.server?.close();
  }
}
```

### 7. Add CI/CD Integration

**.github/workflows/e2e-tests.yml**
```yaml
name: E2E Tests

on: [push, pull_request]

jobs:
  e2e-ios:
    runs-on: macos-latest
    steps:
      - uses: actions/checkout@v3
      - uses: actions/setup-node@v3
        with:
          node-version: 18
      
      - name: Install dependencies
        run: npm ci
      
      - name: Install detox CLI
        run: npm install -g detox-cli
      
      - name: Build app for testing
        run: npm run e2e:build:ios
      
      - name: Run E2E tests
        run: npm run e2e:test:ios
      
      - name: Upload test results
        if: always()
        uses: actions/upload-artifact@v3
        with:
          name: e2e-test-results
          path: e2e/artifacts/

  e2e-android:
    runs-on: ubuntu-latest
    steps:
      # Similar to iOS but for Android
      - uses: actions/checkout@v3
      - uses: actions/setup-node@v3
      - name: Setup Android emulator
        run: |
          echo "y" | $ANDROID_HOME/tools/bin/sdkmanager --install 'system-images;android-33;google_apis;x86_64'
          echo "no" | $ANDROID_HOME/tools/bin/avdmanager create avd -n Pixel_7_API_33 -k 'system-images;android-33;google_apis;x86_64'
      
      - run: npm ci
      - run: npm run e2e:build:android
      - run: npm run e2e:test:android
```

## Technical Considerations
- **Detox vs Appium**: Detox is faster and more reliable for React Native
- **Flaky Tests**: Use \`waitFor\` with timeouts, avoid hardcoded delays
- **Test Data**: Reset database before each test for consistency
- **Environment**: Use separate backend URL for E2E tests
- **Parallelization**: Run tests sequentially (\`maxWorkers: 1\`) to avoid conflicts
- **Screenshots**: Detox automatically captures screenshots on failure
- **CI Performance**: iOS tests require macOS runners (expensive), consider Android-only for PRs
